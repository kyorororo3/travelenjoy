CREATE TABLE TE_TOUR(
    SEQ INT,
    EMAIL VARCHAR(50),
    NICKNAME VARCHAR(100),
    CATEGORY VARCHAR(10),
    TITLE VARCHAR(200),
    CONTENT VARCHAR(1000),
    THUMBNAIL VARCHAR(50),
    PERIOD INT,
    MIN_PEOPLE INT,
    MAX_PEOPLE INT
);

CREATE TABLE TE_TOUR_SCHE(
    SEQ INT,
    TOUR_SEQ INT,
    JOIN_PEOPLE INT,
    START_DATE DATETIME
);

CREATE TABLE TE_TOUR_DES(
    SEQ INT,
    TOUR_SEQ INT,
    DES_NAME VARCHAR(100),
    DES_IMG VARCHAR(50),
    DES_DESCRIPTION VARCHAR(500),
    POSTCODE VARCHAR(10),
    ADDRESS VARCHAR(200),
    ADDRESS_DETAIL VARCHAR(200),
    TOUR_DAY INT,
    START_TIME INT,
    END_TIME INT
);
