CREATE TABLE TE_TOUR(
    SEQ INT AUTO_INCREMENT PRIMARY KEY,
    EMAIL VARCHAR(50),
    NICKNAME VARCHAR(100),
    CATEGORY VARCHAR(10),
    TITLE VARCHAR(200),
    CONTENT VARCHAR(1000),
    THUMBNAIL VARCHAR(50),
    PERIOD INT,
    MIN_PEOPLE INT,
    MAX_PEOPLE INT
);

create function ordered_uuid(uuid binary(36))

begin

end$$
